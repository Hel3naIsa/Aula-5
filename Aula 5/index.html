            import numpy as np
import pandas as pd

def remove_outliers(df, column, threshold=3):
    """
    Remove outliers de uma coluna específica de um DataFrame com base no Z-score.
    
    :param df: DataFrame contendo os dados.
    :param column: Nome da coluna onde os outliers serão removidos.
    :param threshold: O valor do Z-score acima do qual um valor é considerado um outlier (default é 3).
    :return: DataFrame com outliers removidos.
    """
    # Calcula a média e o desvio padrão da coluna
    mean = df[column].mean()
    std_dev = df[column].std()
    
    # Calcula o Z-score para cada valor
    df['z_score'] = (df[column] - mean) / std_dev
    
    # Filtra os dados, mantendo apenas os valores cujo Z-score é menor que o limiar (threshold)
    df_cleaned = df[df['z_score'].abs() <= threshold]
    
    # Remove a coluna 'z_score' que foi criada
    df_cleaned = df_cleaned.drop(columns=['z_score'])
    
    return df_cleaned

# Exemplo de uso com um DataFrame
data = {
    'A': [1, 2, 3, 4, 100, 6, 7, 8, 9, 10]
}
df = pd.DataFrame(data)

df_cleaned = remove_outliers(df, 'A')

print("DataFrame original:")
print(df)
print("\nDataFrame sem outliers:")
print(df_cleaned)
